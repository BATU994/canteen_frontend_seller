# Canteen Admin app

```
src
    ├───core                   # Основная/общая функциональность приложения
    │   ├───api                # Всё, что связано с API (HTTP-запросы, эндпоинты)
    │   │   ├───adress         # Конфигурация адресов/эндпоинтов API
    │   │   └───client         # Настройка HTTP-клиента (например, Dio, http)
    │   ├───const              # Константы (цвета, строки, ключи и т.д.)
    │   ├───enums              # Перечисления (enum) для всего приложения
    │   ├───exceptions         # Пользовательские исключения/обработка ошибок
    │   ├───routers            # Навигация приложения / маршруты
    │   └───service            # Сервисы общего пользования (DI, хранилище и т.д.)
    │       ├───injectable    # Настройка Dependency Injection
    │       └───storage        # Работа с хранилищем
    │           └───remote     # Удалённое хранилище или сетевое
    └───features               # Модули по функциональным направлениям
        ├───orders             # Модуль заказов
        │   ├───data           # Слой данных (модели, репозитории, источники)
        │   │   ├───models      # Модели данных заказов
        │   │   ├───repositories # Реализация репозиториев
        │   │   └───sources     # Источники данных (API, локальная БД)
        │   │       └───remote  # Удалённый источник данных для заказов
        │   ├───domain         # Доменные сущности и бизнес-логика
        │   │   ├───entities    # Сущности (Entities) заказов
        │   │   └───repositories # Абстрактные репозитории
        │   └───presentation   # Слой UI
        │       ├───bloc        # Состояния и управление через Bloc
        │       │   └───orders_main_bloc # Главный Bloc для заказов
        │       ├───pages       # Экраны/страницы заказов
        │       └───widgets     # UI-компоненты для заказов
        └───products           # Модуль продуктов
            ├───data           # Слой данных продуктов
            │   ├───models      # Модели данных продуктов
            │   ├───repositories # Реализация репозиториев продуктов
            │   └───sources     # Источники данных (API, локальная БД)
            ├───domain         # Доменные сущности и логика продуктов
            │   ├───entity      # Сущности (Entities) продуктов
            │   └───repositories # Абстрактные репозитории
            └───presentation   # UI слой для продуктов
                ├───bloc        # Состояния и управление через Bloc
                ├───modals      # Модальные окна, диалоги для продуктов
                ├───pages       # Экраны/страницы продуктов
                └───widgets     # UI-компоненты для продуктов

```
